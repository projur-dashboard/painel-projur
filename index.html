function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("DADOS") || ss.getSheets()[0]; SpreadsheetApp.flush(); var lastRow = sheet.getLastRow(); if (lastRow < 2) return ContentService.createTextOutput(JSON.stringify({"erro": "Vazia"})).setMimeType(ContentService.MimeType.JSON); // Lemos as colunas A até F (index 0 a 5) var rows = sheet.getRange(2, 1, lastRow, 6).getValues(); var dashboardData = { "comPrazo": 0, "geradoEm": Utilities.formatDate(new Date(), "GMT-3", "dd/MM/yyyy HH:mm"), "porAdv": {"JOSINALDO":0, "JULIANA":0, "JULIANNE":0, "RAFAELLE":0}, "agenda": {} }; rows.forEach(function(row) { var advRaw = row[0] ? row[0].toString().trim().toUpperCase() : ""; if (!advRaw || advRaw.includes("RESPONSÁVEL")) return; var advogado = "OUTROS"; if (advRaw.includes("JOSINALDO")) advogado = "JOSINALDO"; else if (advRaw.includes("JULIANNE")) advogado = "JULIANNE"; else if (advRaw.includes("JULIANA")) advogado = "JULIANA"; else if (advRaw.includes("RAFAELLE")) advogado = "RAFAELLE"; if (advogado !== "OUTROS") { dashboardData.porAdv[advogado] = (dashboardData.porAdv[advogado] || 0) + 1; } var tipo = row[1] ? row[1].toString().trim() : ""; var nome = row[2] ? row[2].toString().trim() : ""; var proc = row[3] ? row[3].toString().trim() : ""; // --- LÓGICA ESPECÍFICA PARA A COLUNA F --- var colE = row[4]; // Data com ano (referência) var colF = row[5] ? row[5].toString().trim() : ""; // "04/02 - recurso" var dataFinal = null; var anoReferencia = 2026; // Padrão caso falhe // Tenta extrair o ano da Coluna E if (colE instanceof Date && !isNaN(colE.getTime())) { anoReferencia = colE.getFullYear(); } // Procura o padrão DD/MM no início ou meio da Coluna F var matchF = colF.match(/(\d{1,2})[\/\-](\d{1,2})/); if (matchF) { var dia = parseInt(matchF[1], 10); var mes = parseInt(matchF[2], 10) - 1; // Meses no JS são 0-11 var dTemp = new Date(anoReferencia, mes, dia); if (!isNaN(dTemp.getTime())) { dataFinal = dTemp; } } if (dataFinal) { dashboardData.comPrazo++; var key = Utilities.formatDate(dataFinal, "GMT-3", "yyyy-MM-dd"); if (!dashboardData.agenda[key]) dashboardData.agenda[key] = {}; if (!dashboardData.agenda[key][advogado]) dashboardData.agenda[key][advogado] = []; dashboardData.agenda[key][advogado].push({ nome: nome, proc: proc, tipo: tipo, obs: colF // Mostra o texto completo: "04/02 - recurso" }); } }); // Ordenar agenda por data var sortedKeys = Object.keys(dashboardData.agenda).sort(); var sortedAgenda = sortedKeys.map(function(k) { return { date: k, advs: dashboardData.agenda[k] }; }); dashboardData.agenda = sortedAgenda; return ContentService.createTextOutput(JSON.stringify(dashboardData)).setMimeType(ContentService.MimeType.JSON); }
